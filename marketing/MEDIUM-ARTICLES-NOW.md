// SPDX-License-Identifier: MIT
pragma solidity ^0.8.0;

contract CharityGuardian {
    address public charityWallet;
    address public infrastructureWallet;
    address public founderWallet;

    uint256 public constant CHARITY_PERCENTAGE = 60;
    uint256 public constant INFRASTRUCTURE_PERCENTAGE = 30;
    uint256 public constant FOUNDER_PERCENTAGE = 10;

    event RevenueAllocated(
        uint256 charityShare,
        uint256 infrastructureShare,
        uint256 founderShare,
        uint256 timestamp
    );

    constructor(
        address _charityWallet,
        address _infrastructureWallet,
        address _founderWallet
    ) {
        charityWallet = _charityWallet;
        infrastructureWallet = _infrastructureWallet;
        founderWallet = _founderWallet;
    }

    function allocateRevenue() public payable {
        require(msg.value > 0, "No funds to allocate");

        uint256 charityShare = (msg.value * CHARITY_PERCENTAGE) / 100;
        uint256 infrastructureShare = (msg.value * INFRASTRUCTURE_PERCENTAGE) / 100;
        uint256 founderShare = msg.value - charityShare - infrastructureShare;

        payable(infrastructureWallet).transfer(infrastructureShare);

        emit RevenueAllocated(charityShare, infrastructureShare, founderShare, block.timestamp);
    }
}

**Backend Route:**
```javascript
// api/routes/founding-member.js
router.post('/founding-member/checkout', async (req, res) => {
  const { email, name, displayName, displayPermission } = req.body;

  // Check if spots still available
  const currentCount = await db.foundingMembers.count();
  if (currentCount >= 100) {
    return res.status(400).json({
      error: "All 100 founding member spots have been claimed."
    });
  }

  // Create Square checkout
  const checkout = await square.createCheckout({
    amount: 9900,
    currency: 'USD',
    redirect_url: `${process.env.FRONTEND_URL}/founding-member/welcome`,
    metadata: {
      email,
      name,
      displayName: displayName || name,
      displayPermission: displayPermission || false,
      type: 'founding_member'
    }
  });

  // Reserve spot (pending payment)
  await db.foundingMembers.create({
    email,
    name,
    displayName: displayName || name,
    displayPermission: displayPermission || false,
    status: 'pending',
    checkoutId: checkout.id,
    memberId: `FM${String(currentCount + 1).padStart(3, '0')}` // FM001, FM002, etc.
  });

  res.json({ checkout_url: checkout.url });
});

// Webhook handler for payment confirmation
router.post('/webhook/founding-member/success', async (req, res) => {
  const { checkoutId, paymentId } = req.body;

  // Verify payment with Square
  const payment = await square.getPayment(paymentId);

  if (payment.status === 'COMPLETED') {
    // Update member status
    await db.foundingMembers.update({
      where: { checkoutId },
      data: {
        status: 'active',
        paymentId,
        paidAt: new Date(),
        charityAllocation: 5940,  // $59.40
        infraAllocation: 2970,    // $29.70
        founderAllocation: 990     // $9.90
      }
    });

    // Grant lifetime Pro access
    const member = await db.foundingMembers.findUnique({ where: { checkoutId } });
    await db.users.update({
      where: { email: member.email },
      data: {
        plan: 'founding_member',
        planExpires: null, // Lifetime
        foundingMemberId: member.memberId
      }
    });

    // Send welcome email

    // Update real-time counter
    io.emit('founding-member-joined', {
      memberId: member.memberId,
      displayName: member.displayPermission ? member.displayName : 'Anonymous',
```

### Phase 3: Backend Integration

#### Node.js Implementation (api/routes/email-sequences.js)

```javascript
const sgMail = require('@sendgrid/mail');
const db = require('../db');

sgMail.setApiKey(process.env.SENDGRID_API_KEY);

// Welcome Email 1 (Immediate trigger)
async function sendWelcomeEmail1(user) {
  try {
    const msg = {
      to: user.email,
      from: process.env.SENDGRID_FROM_EMAIL,
      fromname: process.env.SENDGRID_FROM_NAME,
      templateId: 'd-welcome-sequence-01',
      dynamicTemplateData: {
        firstName: user.firstName,
        dashboardUrl: 'https://aidoesitall.website/dashboard',
        referralCode: user.referralCode,
      },
      trackingSettings: {
        clickTracking: { enabled: true },
        openTracking: { enabled: true },
      },
    };

    const response = await sgMail.send(msg);

    // Log email
    await db.emailLog.create({
      userId: user.id,
      templateId: 'd-welcome-sequence-01',
      sequence: 'welcome',
      step: 1,
      sentAt: new Date(),
      messageId: response[0].headers['x-message-id'],
    });

    return response;
  } catch (error) {
    console.error('SendGrid error:', error);
    throw error;
  }
}

// Re-engagement Email 1 (Day 60 trigger)
async function sendReengagementEmail1(userId) {
  const user = await db.users.findById(userId);
  const lastLogin = await db.logins.getLastLogin(userId);

  const daysSinceLogin = Math.floor(
    (Date.now() - lastLogin.timestamp) / (1000 * 60 * 60 * 24)
  );

  if (daysSinceLogin >= 60) {
    const msg = {
      to: user.email,
      from: process.env.SENDGRID_FROM_EMAIL,
      templateId: 'd-reengagement-sequence-01',
      dynamicTemplateData: {
        firstName: user.firstName,
        daysSinceLogin,
        loginUrl: 'https://aidoesitall.website/login',
      },
    };

    await sgMail.send(msg);
  }
}

// Referral Success (Triggered on conversion)
async function sendReferralSuccessEmail(referrerId, referralData) {
  const referrer = await db.users.findById(referrerId);
  const referral = await db.referrals.findById(referralData.id);

  const msg = {
    to: referrer.email,
    from: process.env.SENDGRID_FROM_EMAIL,
    templateId: 'd-purchase-sequence-03',
    dynamicTemplateData: {
      firstName: referrer.firstName,
      friendName: referral.referredUserName,
      referralCount: await db.referrals.getSuccessfulCount(referrerId),
      freeMonthsEarned: await db.referrals.getFreeMonthsEarned(referrerId),
    },
  };

  await sgMail.send(msg);
}

module.exports = {
  sendWelcomeEmail1,
  sendReengagementEmail1,
  sendReferralSuccessEmail,
};
```
